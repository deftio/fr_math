Examples

Short, runnable snippets for the most common FR_Math tasks. Each example compiles cleanly against the v2.0.0 library with:

cc -Isrc example.c src/FR_math.c -lm -o example
./example

The 2D examples additionally need src/FR_math_2D.cpp and a C++ compiler (c++ or g++).

1. Basic radix conversion

#include <stdio.h>
#include "FR_defs.h"
#include "FR_math.h"

int main(void)
{
    /* 3.5 at radix 4 = (3 << 4) + (0.5 * 16) = 48 + 8 = 56 = 0x38 */
    s32 a = I2FR(3, 4) | (1 << 3);       /* 3.5  */
    s32 b = I2FR(2, 4) | (1 << 2);       /* 2.25 */

    printf("a = %d (raw) = %d.%d\n",
           a, FR_FR2I(a, 4), (a & 0xF));

    /* Same 3.5 at radix 8: shift up by 4. */
    s32 a_r8 = FR_CHRDX(a, 4, 8);
    printf("a at radix 8 = %d\n", a_r8);    /* 896 */

    /* Change back to radix 4 (no precision lost because it was <= r8). */
    s32 a_r4 = FR_CHRDX(a_r8, 8, 4);
    printf("round-trip = %d (equals a: %s)\n",
           a_r4, a_r4 == a ? "yes" : "no");
    return 0;
}

2. Trig — integer degrees vs radian/BAM

#include <stdio.h>
#include "FR_math.h"

int main(void)
{
    const int r = 14;

    /* Classic API: integer degrees. */
    s32 s30_deg = FR_Sin(30, r);            /* expect 0.5  -> ~8192  */
    s32 c60_deg = FR_Cos(60, r);            /* expect 0.5  -> ~8192  */

    /* v2 API: BAM (full circle = 65536). 30 degrees = 65536 * 30/360 = 5461. */
    s32 s30_bam = fr_sin_bam((u16)((30L * 65536L + 180) / 360), r);
    s32 c60_bam = fr_cos_bam((u16)((60L * 65536L + 180) / 360), r);

    printf("deg: sin(30) = %d  cos(60) = %d\n", s30_deg, c60_deg);
    printf("bam: sin(30) = %d  cos(60) = %d\n", s30_bam, c60_bam);
    printf("expected     = %d\n", 1 << (r - 1));   /* 0.5 at radix 14 */
    return 0;
}

All four answers should land within ±2 of 1 << 13 = 8192.

3. Square root and hypotenuse

#include <stdio.h>
#include "FR_math.h"

int main(void)
{
    const int r = 16;

    /* sqrt(2) at radix 16 */
    s32 two = I2FR(2, r);
    s32 root = FR_sqrt(two, r);
    printf("sqrt(2) raw = 0x%x\n", root);        /* ~0x00016a09 = 1.41421 */

    /* hypot(3, 4) = 5 exactly */
    s32 three = I2FR(3, r);
    s32 four  = I2FR(4, r);
    s32 h     = FR_hypot(three, four, r);
    printf("hypot(3,4)  = %d.%06d\n",
           FR_FR2I(h, r),
           FR_FR2I((h & 0xFFFF) * 1000000, r));

    /* Domain-error handling */
    s32 bad = FR_sqrt(-1, r);
    if (bad == FR_DOMAIN_ERROR) printf("sqrt(-1) rejected, good.\n");
    return 0;
}

4. Logarithm, exponential, decibels

#include <stdio.h>
#include "FR_math.h"

/* Convert a linear amplitude (radix 16) into dB (also radix 16). */
static s32 amplitude_to_db(s32 amp_r16)
{
    /* 20 * log10(amp) */
    s32 l10 = FR_log10(amp_r16, 16);
    return FR_FixMuls(l10, I2FR(20, 16), 16);
}

int main(void)
{
    s32 half = I2FR(1, 16) >> 1;            /* 0.5 at radix 16 */
    s32 db   = amplitude_to_db(half);        /* expect ~ -6.02 dB */
    printf("0.5 -> %d dB (radix 16)\n", FR_FR2I(db, 16));

    /* Invert: 10^(-6 / 20) should give us roughly 0.5 back. */
    s32 x = FR_pow10(FR_FixMuls(I2FR(-6, 16), I2FR(1, 16) / 20, 16), 16);
    printf("10^(-6/20) = 0x%x (expect ~0x8000)\n", x);
    return 0;
}

5. Arctangent and atan2

#include <stdio.h>
#include "FR_math.h"

int main(void)
{
    const int r = 14;

    /* atan(1) = pi/4 */
    s32 a = FR_atan(I2FR(1, r), r);
    printf("atan(1) raw = %d (expect ~%d)\n",
           a, (int)(3.14159265 / 4.0 * (1 << r)));

    /* Full-quadrant using atan2 */
    s32 q2 = FR_atan2(I2FR( 1, r), I2FR(-1, r), r);  /* 3pi/4  */
    s32 q3 = FR_atan2(I2FR(-1, r), I2FR(-1, r), r);  /* -3pi/4 */
    printf("atan2( 1,-1) = %d\n", q2);
    printf("atan2(-1,-1) = %d\n", q3);

    /* Error sentinel: atan2(0,0) is undefined */
    if (FR_atan2(0, 0, r) == FR_DOMAIN_ERROR)
        printf("atan2(0,0) rejected, good.\n");
    return 0;
}

6. Wave generators + phase accumulator

#include <stdio.h>
#include "FR_math.h"

/* Generate 16 samples of a 1 kHz triangle at 48 kHz sample rate. */
int main(void)
{
    const u32 SR   = 48000;
    const u32 FREQ = 1000;

    /* Phase step in BAM per sample: (2^16 * FREQ / SR) */
    u16 step = (u16)((65536ULL * FREQ) / SR);

    u16 phase = 0;
    for (int i = 0; i < 16; i++) {
        s16 tri = fr_wave_tri(phase);
        s16 sqr = fr_wave_sqr(phase);
        printf("%2d: phase=%5u  tri=%6d  sqr=%6d\n", i, phase, tri, sqr);
        phase += step;            /* wraps automatically at 2^16 */
    }
    return 0;
}

The wave functions are stateless, so you can drive as many voices as you like from independent phase accumulators.

7. ADSR envelope

#include <stdio.h>
#include "FR_math.h"

int main(void)
{
    fr_adsr_t env;
    fr_adsr_init(&env,
                 /*attack  */ 100,
                 /*decay   */ 200,
                 /*sustain */ FR_WAVE_PEAK / 2,
                 /*release */ 400);

    fr_adsr_trigger(&env);

    /* Step through 500 samples of sustain */
    for (int i = 0; i < 500; i++) {
        s16 y = fr_adsr_step(&env);
        if (i % 50 == 0) printf("t=%3d  amp=%d\n", i, y);
    }

    fr_adsr_release(&env);

    for (int i = 0; i < 500; i++) {
        s16 y = fr_adsr_step(&env);
        if (i % 50 == 0) printf("rel t=%3d  amp=%d\n", i, y);
    }
    return 0;
}

8. 2D transforms

#include <stdio.h>
#include "FR_math.h"
#include "FR_math_2D.h"

int main()
{
    /* Build a TRS transform: translate * rotate(30) * scale(2) */
    FR_Matrix2D_CP T, R, S, M;
    T.setidentity(); T.settranslate(I2FR(100, 16), I2FR(50, 16));
    R.setidentity(); R.setrotate(30);
    S.setidentity(); S.setscale(I2FR(2, 16), I2FR(2, 16));

    M = T; M.mul(R); M.mul(S);

    FR_2DPoint p = { I2FR(10, 16), I2FR(0, 16) };
    M.XformPt(&p);

    printf("transformed point = (%d.%03d, %d.%03d)\n",
           FR_FR2I(p.x, 16),
           FR_FR2I((p.x & 0xFFFF) * 1000, 16),
           FR_FR2I(p.y, 16),
           FR_FR2I((p.y & 0xFFFF) * 1000, 16));

    /* Invert and map the point back to (10, 0). */
    if (M.inv()) {
        M.XformPt(&p);
        printf("round trip = (%d, %d)\n",
               FR_FR2I(p.x, 16), FR_FR2I(p.y, 16));
    }
    return 0;
}

9. Formatted output without printf

#include "FR_math.h"

/* Collect output into a local buffer so we don't need stdio. */
static char buf[64];
static int  cursor = 0;

static void sink(char c)
{
    if (cursor < (int)sizeof(buf) - 1) buf[cursor++] = c;
}

int main(void)
{
    s32 x = (s32)(3.14159 * (1 << 16));     /* pi at radix 16 */
    FR_printNumF(x, 16, 6, sink);             /* "3.141586" */
    sink('\n');
    sink('\0');

    /* buf now contains the rendered number, no printf used. */
    return 0;
}

10. Integer-only 2D transform for scanline renderers

#include "FR_math.h"
#include "FR_math_2D.h"

/* XformPtI uses an integer-only point type. Slightly lossier but
   avoids the fixed-point conversion on the hot path — useful
   inside an inner rasterisation loop. */
int main()
{
    FR_Matrix2D_CP M;
    M.setidentity();
    M.setrotate(45);

    FR_2DPointI p = { 100, 0 };     /* plain ints */
    M.XformPtI(&p);                 /* p is now ~(71, 71) */
    return 0;
}
